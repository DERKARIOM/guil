#!/bin/bash
declare -A tab
declare -i i=0
source /usr/local/src/guilh.sh
# x.gtkmm
if [ "$1" == "-v" -o "$1" == "--version" ] ; then
    version
elif [ "$1" == "-h" -o "$1" == "--help" ] ; then
    help
else
    cherche "$@"
    shift "$?"
    if [ "$1" == "-o" ] ; then
        g++ "${tab[@]}" -o "$2" $(pkg-config gtkmm-3.0 --cflags --libs)
    elif [ "$1" == "-ox" ] ; then
        clear ; g++ "${tab[@]}" -o "$2" $(pkg-config gtkmm-3.0 --cflags --libs) && ./"$2"
    fi
fi
